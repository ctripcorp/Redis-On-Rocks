  ---
  设计方案：Storage Config 注册机制

  现有架构分析

  ┌─────────────────────────────────┐
  │ src/config.c                    │
  │  - standardConfig (内部类型)     │
  │  - createBoolConfig 等宏 (内部)  │
  │  - static_configs[] 数组         │
  │  - initConfigValues() 注册入口   │
  │  - registerConfigValue() 内部函数│
  └────────────┬────────────────────┘
               │ server.storage.xxx
               ▼
  ┌─────────────────────────────────┐
  │ deps/ctrip_storage/             │
  │  ctrip_storage_types.h          │
  │   └ StorageServerNamespace      │
  │      - swap_absent_cache_enabled│
  │      - swap_threads_auto_scale_*│
  │      - swap_persist_enabled     │
  │      - ... 30+ 字段             │
  └─────────────────────────────────┘

  核心约束：standardConfig、createBoolConfig 等宏都定义在 config.c
  内部，不对外暴露。ctrip_storage 无法直接使用这些宏。

  ---
  方案对比

  ┌──────────┬─────────────────────────────────┬─────────────────────────────┐
  │          │ 方案 A：直接注入 static_configs │ 方案 B：描述符表 + 注册函数 │
  ├──────────┼─────────────────────────────────┼─────────────────────────────┤
  │ 复杂度   │ 低                              │ 中                          │
  ├──────────┼─────────────────────────────────┼─────────────────────────────┤
  │ 模块化   │ 弱（config.c 需知道每个字段）   │ 强（ctrip_storage 自治）    │
  ├──────────┼─────────────────────────────────┼─────────────────────────────┤
  │ 新增配置 │ 改 config.c                     │ 只改 ctrip_storage          │
  ├──────────┼─────────────────────────────────┼─────────────────────────────┤
  │ 风格     │ Redis 原生模式                  │ 类似 Module Config 模式     │
  └──────────┴─────────────────────────────────┴─────────────────────────────┘

  推荐方案 B：ctrip_storage 作为独立 deps 库，应该自己管理配置定义，config.c
  只负责翻译注册。

  方案 B 详细设计

  1. 新增文件：deps/ctrip_storage/ctrip_storage_config.h

  定义一个轻量级的配置描述符，不依赖 config.c 内部类型：

  #ifndef __CTRIP_STORAGE_CONFIG_H__
  #define __CTRIP_STORAGE_CONFIG_H__

  #include <stddef.h>

  /* 存储配置类型枚举 */
  typedef enum {
      STORAGE_CONF_BOOL,       /* int, 0/1 */
      STORAGE_CONF_INT,        /* int, 有上下界 */
      STORAGE_CONF_LONG_LONG,  /* long long, 有上下界 */
      STORAGE_CONF_STRING,     /* char* */
  } StorageConfType;

  /* 配置项标志位（与 config.c 中的 flags 对应） */
  #define STORAGE_CONF_IMMUTABLE   (1 << 0)  /* 启动后不可修改 */
  #define STORAGE_CONF_MODIFIABLE  (1 << 1)  /* 运行时可 CONFIG SET */
  #define STORAGE_CONF_HIDDEN      (1 << 2)  /* 隐藏配置 */

  /* 配置描述符 - ctrip_storage 自己定义的配置元信息 */
  typedef struct StorageConfigDesc {
      const char *name;          /* 配置项名称，如 "swap-absent-cache-enabled" 
  */
      const char *alias;         /* 别名，可为 NULL */
      StorageConfType type;      /* 配置值类型 */
      int flags;                 /* STORAGE_CONF_xxx 标志 */
      size_t offset;             /* offsetof(StorageServerNamespace, field) */
      union {
          struct { int default_val; } bval;                              /* BOOL
   */
          struct { int default_val; int lower; int upper; } ival;        /* INT 
  */
          struct { long long default_val; long long lower; long long upper; }
  llval; /* LONG_LONG */
          struct { const char *default_val; int empty_to_null; } sval;   /* 
  STRING */
      } def;
  } StorageConfigDesc;

  /*
   * 获取存储配置描述符表
   * 输入: count - 输出参数，返回配置项数量
   * 输出: 指向静态描述符数组的指针
   */
  const StorageConfigDesc *storageGetConfigDescriptors(int *count);

  #endif

  3. 修改 src/config.c：翻译注册

  在 initConfigValues() 中调用翻译注册函数：

  #include "ctrip_storage_config.h"

  /*
   * 将 StorageConfigDesc flags 转换为 config.c 内部 flags
   */
  static unsigned int storageConfFlagsToConfigFlags(int storage_flags) {
      unsigned int flags = 0;
      if (storage_flags & STORAGE_CONF_IMMUTABLE)  flags |= IMMUTABLE_CONFIG;
      if (storage_flags & STORAGE_CONF_MODIFIABLE) flags |= MODIFIABLE_CONFIG;
      if (storage_flags & STORAGE_CONF_HIDDEN)     flags |= HIDDEN_CONFIG;
      return flags;
  }

  /*
   * 注册所有 ctrip_storage 定义的配置项
   * 从 StorageConfigDesc 描述符表翻译为 standardConfig 并注册到 configs 字典
   */
  static void registerStorageConfigs(void) {
      int count = 0;
      const StorageConfigDesc *descs = storageGetConfigDescriptors(&count);
      /* 通过 server.storage 基地址 + offset 计算配置字段的实际地址 */
      char *base = (char *)&server.storage;

      for (int i = 0; i < count; i++) {
          const StorageConfigDesc *d = &descs[i];
          unsigned int flags = storageConfFlagsToConfigFlags(d->flags);
          standardConfig sc;

          switch (d->type) {
          case STORAGE_CONF_BOOL: {
              int *addr = (int *)(base + d->offset);
              sc = createBoolConfig(d->name, d->alias, flags,
                                    *addr, d->def.bval.default_val, NULL, NULL);
              sc.data.yesno.config = addr;  /* 修正地址指向实际字段 */
              break;
          }
          case STORAGE_CONF_INT: {
              int *addr = (int *)(base + d->offset);
              sc = createIntConfig(d->name, d->alias, flags,
                                   d->def.ival.lower, d->def.ival.upper,
                                   *addr, d->def.ival.default_val,
                                   INTEGER_CONFIG, NULL, NULL);
              sc.data.numeric.config.i = addr;
              break;
          }
          case STORAGE_CONF_LONG_LONG: {
              long long *addr = (long long *)(base + d->offset);
              sc = createLongLongConfig(d->name, d->alias, flags,
                                        d->def.llval.lower, d->def.llval.upper,
                                        *addr, d->def.llval.default_val,
                                        INTEGER_CONFIG, NULL, NULL);
              sc.data.numeric.config.ll = addr;
              break;
          }
          case STORAGE_CONF_STRING: {
              char **addr = (char **)(base + d->offset);
              sc = createStringConfig(d->name, d->alias, flags,
                                      d->def.sval.empty_to_null,
                                      *addr, d->def.sval.default_val,
                                      NULL, NULL);
              sc.data.string.config = (char **)addr;
              break;
          }
          }

          /* 初始化默认值 */
          if (sc.interface.init) sc.interface.init(&sc);
          /* 注册到全局配置字典 */
          registerConfigValue(sc.name, &sc, 0);
          if (sc.alias) {
              registerConfigValue(sc.alias, &sc, ALIAS_CONFIG);
          }
      }
  }

  /* 修改 initConfigValues，在静态配置注册后调用 */
  void initConfigValues(void) {
      configs = dictCreate(&sdsHashDictType);
      dictExpand(configs, sizeof(static_configs) / sizeof(standardConfig));
      for (standardConfig *config = static_configs; config->name != NULL;
  config++) {
          if (config->interface.init) config->interface.init(config);
          int ret = registerConfigValue(config->name, config, 0);
          serverAssert(ret);
          if (config->alias) {
              int ret = registerConfigValue(config->alias, config,
  ALIAS_CONFIG);
              serverAssert(ret);
          }
      }

      /* 注册 ctrip_storage 定义的配置项 */
      registerStorageConfigs();
  }

  4. 调用链路总览

  Redis 启动
    └─ main()
         └─ initConfigValues()                    [src/config.c]
              ├─ 遍历 static_configs[] 注册原生配置
              └─ registerStorageConfigs()          [src/config.c]
                   └─ storageGetConfigDescriptors() [deps/ctrip_storage]
                        └─ 返回描述符表
                   └─ 遍历描述符，用 offsetof + &server.storage 计算地址
                   └─ 创建 standardConfig 并 registerConfigValue()

  CONFIG SET swap-absent-cache-enabled yes
    └─ configSetCommand()
         └─ lookupConfig("swap-absent-cache-enabled")
              └─ 找到已注册的 standardConfig
         └─ boolConfigSet() → 直接写 server.storage.swap_absent_cache_enabled

  CONFIG GET swap-absent-cache-enabled
    └─ configGetCommand()
         └─ boolConfigGet() → 直接读 server.storage.swap_absent_cache_enabled

  ---
  关键设计决策

  ┌─────────────────────────────┬───────────────────────────────────────────┐
  │            决策             │                   理由                    │
  ├─────────────────────────────┼───────────────────────────────────────────┤
  │ 用 offsetof 而非直接地址    │ 描述符表是 static const，编译期不知道     │
  │                             │ server 地址                               │
  ├─────────────────────────────┼───────────────────────────────────────────┤
  │ flags 独立定义而非复用      │ ctrip_storage 不应依赖 config.c 内部宏    │
  │ config.c                    │                                           │
  ├─────────────────────────────┼───────────────────────────────────────────┤
  │ 描述符表在 ctrip_storage 中 │ 新增配置只改 deps/，不改 src/             │
  ├─────────────────────────────┼───────────────────────────────────────────┤
  │ 翻译层在 config.c 中        │ standardConfig 和宏仍保持 config.c 内部   │
  └─────────────────────────────┴───────────────────────────────────────────┘

  需要注意的问题

  1. createXxxConfig 宏用了 &(config_addr)：宏展开时取地址，但我们用 offsetof
  计算的动态地址需要在宏展开后覆盖 config 指针（如 sc.data.yesno.config = addr）
  2. apply 回调：如果某些配置修改后需要触发动作（如线程池扩缩容），可以在
  StorageConfigDesc 中增加一个 apply_fn 字段，由 ctrip_storage 实现回调函数
  3. 编译依赖：src/config.c 需要 #include "ctrip_storage_config.h"，在 Makefile
  中确保 include path 包含 deps/ctrip_storage/
